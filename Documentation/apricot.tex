\newcommand{\PressureRK}
{
\begin{displaymath}
	P = \frac{RT}{V_{m}-b} - \frac{a}  {\sqrt{T} V_{m} (V_{m} +b)}
\end{displaymath}

where:
\begin{itemize}
    \item P: is the gas pressure,
    \item R: is the gas constant,
    \item T: is the gas temperature,
    \item $V_{m}$ (V/$N_{G}$): is the gas molar volume (where V is volume and $N_{G}$ is moles of fission gas),
    \item a:  is a constant that corrects for attractive potential of molecules,
    \item b:  is a constant that corrects for volume.
\end{itemize}
}

\newcommand{\NumDenFGs}
{
\begin{displaymath}
	N_{G} = \rho_{f}*V_{f}*\frac{Bu}{100}*\gamma_{G}*f*\frac{X^{U}}{M^{U}}
\end{displaymath}

where:
\begin{itemize}
    \item $\rho_{f}$: is the density of fuel material,
    \item $V_{f}$: is the volume of fuel,
    \item $Bu$: is the percentage burnup of fissile and fertile atoms originally present,
    \item $\gamma_{G}$: is the total yield of gaseous fission products (gas atoms produced per fission),
    \item $f$: is the fraction of fission gas released from fuel (atoms released per atom formed),
    \item $X^{U}$: is the mass fraction of uranium in fuel,
    \item $M^{U}$: is the atomic weight of uranium.
    
\end{itemize}
}

\newcommand{\StrainElastic}
{
\begin{displaymath}
 \varepsilon_{r}^{el} = \frac{1}{E} (\sigma_{r} -2 \mu \sigma_{\theta}), \: \:
 \varepsilon_{\theta}^{el} = \frac{1}{E}((1-\mu) \sigma_{\theta} -\mu \sigma_{r})   \: \: \: (3)
\end{displaymath}
where:
\begin{itemize}
	\item $ \varepsilon_{r}^{el} $: is the radial elastic strain,
	\item $ \varepsilon_{\theta}^{el} $: is the tangential elastic strain,
	\item $ \sigma_{r} $: is the radial stress,
	\item $\sigma_{\theta} $: is the tangential stress,
    \item $E$: is the Young's module,
    \item $ \mu $: is the Poisson's ratio.
\end{itemize}
}

\end{displaymath}
where:
\begin{itemize}
	\item $\varepsilon_{r}^{cr}$: is the radial creep strain,
	\item $\varepsilon_{\theta}^{el}$: is the tangential creep strain,
	\item $\sigma_{r}$: is the radial stress,
	\item $\sigma_{\theta}$: is the tangential stress,
    \item $K$: is the irradiation creep coefficient,
    \item $\nu$: is the Poisson's ratio in irradiation creep,
    \item $\Phi$: is the fast neutron fluence.
\end{itemize}
}
